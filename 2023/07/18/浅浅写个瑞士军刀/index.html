  1. import argparse：导入argparse模块，用于解析命令行参数。
  2. import socket：导入socket模块，用于创建套接字和网络通信。
  3. import shlex：导入shlex模块，用于解析字符串成shell单词列表。
  4. import subprocess：导入subprocess模块，用于执行外部命令。
  5. import sys：导入sys模块，用于访问系统相关的功能。
  6. import textwrap：导入textwrap模块，用于格式化文本。
  7. def execute(cmd):：定义了一个名为execute的函数，用于执行给定的命令。
  8. cmd = cmd.strip()：去除命令字符串两端的空白字符。
  9. if not cmd:：如果命令为空，则返回。
  10. output = subprocess.check_output(shlex.split(cmd), stderr=subprocess.STDOUT)：使用subprocess模块的check_output函数执行命令，并将结果存储在output变量中。
  11. return output.decode()：将output变量解码成字符串并返回。
  12. class NetCat:：定义了一个名为NetCat的类，用于实现网络通信功能。
  13. def init(self, args, buffer=None):：NetCat类的初始化方法，接收args和buffer参数。
  14. self.args = args：将参数args赋值给NetCat对象的args属性。
  15. self.buffer = buffer：将参数buffer赋值给NetCat对象的buffer属性。
  16. self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)：创建一个TCP套接字对象，并将其赋值给NetCat对象的socket属性。
  17. self.socket.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)：设置套接字选项，以便允许在套接字关闭后重新使用相同的地址。
  18. def run(self):：定义了一个名为run的方法，用于根据命令行参数执行不同的操作。
  19. if self.args.listen:：检查是否设置了监听模式。
  20. self.listen()：调用listen方法开始监听操作。
  21. self.send()：调用send方法开始发送操作。
  22. def send(self):：定义了一个名为send的方法，用于向目标主机发送数据。
  23. self.socket.connect((self.args.target, self.args.port))：和目标主机建立TCP连接。
  24. if self.buffer:：检查是否有输入数据。
  25. self.socket.send(self.buffer)：发送输入数据到目标主机。
  26. while True:：无限循环，用于持续接收和发送数据。
  27. recv_len = 1：初始化recv_len变量为1。
  28. response = ‘’：初始化response变量为空字符串。
  29. while recv_len:：循环接收数据，直到接收长度为0（连接关闭）。
  30. data = self.socket.recv(4096)：接收最多4096字节的数据。
  31. recv_len = len(data)：获取接收到的数据的长度。
  32. response += data.decode()：将接收到的数据解码并添加到response变量中。
  33. if recv_len < 4096:：如果接收到的数据长度小于4096，则退出循环。
  34. break：跳出内部循环。
  35. if response:：检查是否接收到了数据。
  36. print(response)：打印接收到的数据。
  37. buffer = input(‘> ‘)：从用户输入获取命令。
  38. buffer += ‘\n’：在命令末尾添加换行符。
  39. self.socket.send(buffer.encode())：将命令发送给目标主机。
  40. except KeyboardInterrupt:：捕获键盘中断异常。
  41. print(‘User terminated.’)：打印用户终止信息。
  42. self.socket.close()：关闭套接字连接。
  43. sys.exit()：退出程序。
  44. def listen(self):：定义了一个名为listen的方法，用于开始监听连接请求。
  45. print(‘listening’)：打印监听信息。
  46. self.socket.bind((self.args.target, self.args.port))：将套接字绑定到指定的IP地址和端口。
  47. self.socket.listen(5)：开始监听连接请求，允许同时最多5个连接。
  48. while True:：无限循环，用于持续处理连接。
  49. client_socket, _ = self.socket.accept()：接受客户端连接，返回客户端套接字对象和地址信息。
  50. client_thread = threading.Thread(target=self.handle, args=(client_socket,))：创建一个新线程，用于处理客户端连接。
  51. client_thread.start()：启动新线程。
  52. def handle(self, client_socket):：定义了一个名为handle的方法，用于处理客户端连接。
  53. if self.args.execute:：检查是否设置了执行命令参数。
  54. output = execute(self.args.execute)：执行指定的命令，并将结果赋值给output变量。
  55. client_socket.send(output.encode())：将命令执行结果发送回客户端。
  56. elif self.args.upload:：检查是否设置了上传文件参数。
  57. file_buffer = b’’：初始化file_buffer变量为空字节串。
  58. while True:：无限循环，用于持续接收文件数据。
  59. data = client_socket.recv(4096)：接收最多4096字节的文件数据。
  60. if data:：检查是否接收到了数据。
  61. file_buffer += data：将接收到的数据添加到file_buffer变量中。
  62. print(len(file_buffer))：打印已接收文件数据的长度。
  63. else:：如果未接收到数据，跳出循环。
  64. break：跳出内部循环。
  65. with open(self.args.upload, ‘wb’) as f:：以二进制写入模式打开要上传的文件。
  66. f.write(file_buffer)：将接收到的文件数据写入文件。
  67. message = f’Saved file {self.args.upload}’：构造上传成功的消息。
  68. client_socket.send(message.encode())：将上传成功的消息发送回客户端。
  69. elif self.args.command:：检查是否设置了传输命令参数。
  70. cmd_buffer = b’’：初始化cmd_buffer变量为空字节串，用于存储传输的命令。
  71. while True:：无限循环，用于持续传输命令。
  72. client_socket.send(b’ #> ‘)：发送命令提示符给客户端。
  73. while ‘\n’ not in cmd_buffer.decode():：循环直到接收到完整的命令。
  74. cmd_buffer += client_socket.recv(64)：接收最多64字节的命令数据。
  75. response = execute(cmd_buffer.decode())：执行接收到的命令，并将结果赋值给response变量。
  76. if response:：当命令执行结果非空时。
  77. client_socket.send(response.encode())：将命令执行结果发送给客户端。
  78. cmd_buffer = b’’：清空cmd_buffer变量。
  79. print(f’server killed {e}’)：打印服务器被终止的错误信息。
  80. self.socket.close()：关闭套接字连接。
  81. sys.exit()：退出程序。
  82. if name == ‘main‘:：如果当前模块是主程序入口。
  83. parser = argparse.ArgumentParser(…)：创建一个命令行参数解析器。
  84. parser.add_argument(…)：添加命令行参数选项。
  85. args = parser.parse_args()：解析命令行参数，并将结果赋值给args变量。
  86. if args.listen:：检查是否设置了监听模式。

  88. else:：如果不是监听模式。
